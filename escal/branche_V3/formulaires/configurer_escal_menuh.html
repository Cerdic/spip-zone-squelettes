Choisir la présence ou l'absence d'un menu sur toutes les pages du site.

  • #SET{name,menuH}#SET{erreurs,#ENV**{erreurs}|table_valeur{#GET{name}}} Affichage du menu horizontal [(#GET{erreurs}) ]

    #SET{name,articlesmenuH}#SET{erreurs,#ENV**{erreurs}|table_valeur{#GET{name}}} Avec affichage des articles [(#GET{erreurs}) ]

    Pour les rubriques de premier niveau (= secteur) :
    #SET{name,texterubmenu}#SET{erreurs,#ENV**{erreurs}|table_valeur{#GET{name}}} Affichage du titre de la rubrique [(#GET{erreurs}) ]

    #SET{name,logorubmenu}#SET{erreurs,#ENV**{erreurs}|table_valeur{#GET{name}}} Affichage du logo de la rubrique s'il existe [(#GET{erreurs}) ]

    #SET{name,taillelogorubmenu}#SET{erreurs,#ENV**{erreurs}|table_valeur{#GET{name}}} Hauteur des logos de rubrique en pixels [(#GET{erreurs}) ]
    Veillez à ce que vos logos aient une hauteur au moins égale à la valeur choisie

    #SET{name,menufixe}#SET{erreurs,#ENV**{erreurs}|table_valeur{#GET{name}}} Fixer le menu en haut de page lors du scroll vers le bas [(#GET{erreurs}) ]
